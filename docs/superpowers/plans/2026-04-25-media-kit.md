# Escape Velocity AI Media Kit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a credibility document at `escape-velocity-ai.com/media-kit` (web + downloadable PDF) for the Escape Velocity AI agency.

**Architecture:** Astro static site, hybrid presentation (single-scroll web + dedicated print route + pre-generated PDF via Playwright). Data-file-driven content so a single edit flips the channel from pre-launch to launched without redesign. Deployed on Cloudflare Pages with custom domain.

**Tech Stack:** Astro 4, TypeScript, plain CSS (no Tailwind — token-driven), Playwright (headless Chromium for PDF generation), Cloudflare Pages.

**Spec reference:** `docs/superpowers/specs/2026-04-25-media-kit-design.md`

```astro
  // Channel stats with localStorage-backed click-to-edit.
  // - Reads `evai-channel-stats` on load and overrides server-rendered values.
  // - Toggles section visibility based on combined values (server defaults OR local overrides).
  // - Clicking a stat prompts for a new value; saved per-browser.
  const STORAGE_KEY = "evai-channel-stats";

  type StatKey = "subscribers" | "videosPublished" | "totalViews";

  function loadOverrides(): Partial<Record<StatKey, number>> {
    try {
      return JSON.parse(localStorage.getItem(STORAGE_KEY) || "{}");
    } catch {
      return {};
    }
  }

  function saveOverrides(overrides: Partial<Record<StatKey, number>>) {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(overrides));
  }

  function format(n: number): string {
    return new Intl.NumberFormat("en-US").format(n);
  }

  function applyAndToggle() {
    const section = document.querySelector<HTMLElement>("[data-channel-section]");
    if (!section) return;

    const overrides = loadOverrides();
    const stats = section.querySelectorAll<HTMLElement>("[data-stat]");
    let anyNonZero = false;

    stats.forEach((el) => {
      const key = el.dataset.stat as StatKey;
      const fallback = Number(el.dataset.default || 0);
      const value = key in overrides ? Number(overrides[key]) : fallback;
      const numEl = el.querySelector<HTMLElement>(".number");
      if (numEl) numEl.textContent = format(value);
      if (value > 0) anyNonZero = true;
    });

    section.style.display = anyNonZero ? "" : "none";
  }

  function bindClickEditing() {
    document.querySelectorAll<HTMLElement>("[data-stat]").forEach((el) => {
      el.addEventListener("click", () => {
        const key = el.dataset.stat as StatKey;
        const current = el.querySelector(".number")?.textContent || "0";
        const input = window.prompt(
          `Enter new value for "${key}" (saved to this browser only):`,
          current.replace(/,/g, "")
        );
        if (input === null) return;
        const num = Number(input.replace(/[^0-9]/g, ""));
        if (Number.isNaN(num)) return;
        const overrides = loadOverrides();
        overrides[key] = num;
        saveOverrides(overrides);
        applyAndToggle();
      });
    });
  }

  applyAndToggle();
  bindClickEditing();
</script>
```

```js
import { chromium } from "playwright";

  process.exit(1);
}

console.log("Starting Astro preview server...");
const server = spawn("npx", ["astro", "preview", "--port", String(PORT)], {
  stdio: ["ignore", "pipe", "pipe"],
  shell: process.platform === "win32",
});

server.stdout.on("data", (chunk) => process.stdout.write(`[astro] ${chunk}`));
server.stderr.on("data", (chunk) => process.stderr.write(`[astro] ${chunk}`));

// Wait for the server to be reachable (poll up to 30s)
async function waitForServer() {
  for (let i = 0; i < 60; i++) {
    try {
      const res = await fetch(BASE);
      if (res.ok) return true;
    } catch {}
    await wait(500);
  }
  return false;
}

const ready = await waitForServer();
if (!ready) {
  console.error("Astro preview never became ready.");
  server.kill();
  process.exit(1);
}

console.log("Launching Chromium...");
const browser = await chromium.launch();
const page = await browser.newPage();

console.log(`Navigating to ${PRINT_URL}...`);
await page.goto(PRINT_URL, { waitUntil: "networkidle" });

// Force-open all <details> for FAQ
await page.evaluate(() => {
  document.querySelectorAll("details").forEach((d) => (d.open = true));
});

console.log(`Rendering PDF → ${OUTPUT_PATH}`);
await page.pdf({
  path: OUTPUT_PATH,
  format: "Letter",
  margin: { top: "0.75in", right: "0.75in", bottom: "0.85in", left: "0.75in" },
  printBackground: true,
  displayHeaderFooter: true,
  headerTemplate: '<div></div>',
  footerTemplate: `
    <div style="font-size:9pt;color:#6B7280;width:100%;padding:0 0.75in;display:flex;justify-content:space-between;">
      <span>Escape Velocity AI · Media Kit</span>
      <span>Page <span class="pageNumber"></span> of <span class="totalPages"></span></span>
    </div>
  `,
});

await browser.close();
server.kill();

console.log("Done.");
```
